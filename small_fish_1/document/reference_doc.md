Alternatively if it is just an initialisation script the following could be used.

  create_table :posts do |t|
        t.string  :title

   end

   create_table :people do |t|
      t.string :first_name
      t.string :last_name
      t.string :short_name
   end

   create_table :tags do |t|

   end 
end
